\begin{frame}{GRA Meta-zeroing: A Mathematical Framework for Consistency}

\begin{block}{Core Idea}
Recursively eliminate conflicts between all levels using \textbf{foam} – a measure of inconsistency.
\end{block}

\vspace{0.5cm}

\begin{columns}[T]
\column{0.45\textwidth}
\textbf{Key Concepts:}
\begin{itemize}
    \item \textbf{Multi-indices}: $(robot, motor, planner, ethics)$
    \item \textbf{Goals as Projectors}: $P_{G_l}|\psi\rangle = |\psi\rangle$
    \item \textbf{Foam}: $\Phi^{(l)} = \sum_{a\neq b} |\langle\psi_a|P_{G_l}|\psi_b\rangle|^2$
    \item \textbf{Zeroing Theorem}: $\exists \Psi^*$ s.t. $\Phi^{(l)}=0\;\forall l$
\end{itemize}

\column{0.55\textwidth}
\begin{align*}
\Phi^{(l)} &= \sum_{a\neq b} |\langle\psi_a|P_{G_l}|\psi_b\rangle|^2 \\[0.3cm]
J_{\text{total}} &= \sum_{l}\Lambda_l\sum_{\dim(a)=l} J^{(l)}(\psi_a) \\[0.3cm]
J^{(l)} &= \sum_{b\prec a} J^{(l-1)}(\psi_b) + \Phi^{(l)}
\end{align*}
\end{columns}

\vspace{0.5cm}
\centering
\onslide<2>{\textbf{Theorem:} Under commutativity and consistency, full zeroing is achievable.}
\end{frame}

\begin{frame}{Appendix: Mathematical Formulation}

\textbf{Foam:}
$$\Phi^{(l)}(\Psi^{(l)}, G_l) = \sum_{\substack{a\neq b \\ \dim(a)=\dim(b)=l}} |\langle \Psi^{(a)} | \mathcal{P}_{G_l} | \Psi^{(b)} \rangle|^2$$

\textbf{Total Functional:}
$$J_{\text{total}}(\mathbf{\Psi}) = \sum_{l=0}^K \Lambda_l \sum_{\dim(a)=l} J^{(l)}(\Psi^{(a)})$$

\textbf{Recursive Definition:}
$$J^{(0)}(\Psi^{(a)}) = J_{\text{loc}}(\Psi^{(a)}; G_0^{(a)})$$
$$J^{(l)}(\Psi^{(a)}) = \sum_{b\prec a} J^{(l-1)}(\Psi^{(b)}) + \Phi^{(l)}(\Psi^{(a)}, G_l^{(a)})$$

\textbf{Zeroing Theorem:} If projectors commute and hierarchy is consistent, $\exists \Psi^*$ with $\Phi^{(l)}=0$ for all $l$.
\end{frame}
